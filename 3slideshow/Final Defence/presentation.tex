		\begin{itemize}
			\item „Režim soběstačného vykreslování“
			\item V inicializační fázi se definují jednotlivé ovládací prvky a na ně se navěšují callback metody
			\item Hlavní smyčka pak řeší vykreslování prvků a reakce na události, před programátorem je skryta
			\item Knihovny \textit{Qt}, \textit{GTK}, \textit{wxWidgets}, \textit{WPF}, \dots
		\end{itemize}

	\begin{frame}[plain]\frametitle{IMGUI – Immediate mode GUI}
		\begin{itemize}
			\item „Režim okamžitého vykreslování“
			\item Žádná inicializační fáze ani callback metody, pouze hlavní smyčka			
			\item Knihovny \textit{Dear ImGui}, \textit{egui}, \textit{Nuklear}, \dots
		\end{itemize}
		\lstinputlisting[language=pseudo]{code/imgui.txt}
		\begin{itemize}
			\item Definice zmiňuje pouze tento rozdíl z pohledu programátora koncové aplikace, implementace knihoven mohou být různé.
			\item V praxi ale:
			\begin{itemize}
				\item Využití grafického API pro vykreslování ovládacích prvků (\textit{OpenGL}, \textit{DirectX}, \dots)
				\item Časté použití pro přidání ovládacích prvků do existující interaktivní aplikace
				\item Spíše GUI pro ladění vývojářem, než user-facing GUI
				\item Chaos ohledně toho, co je a není IMGUI
			\end{itemize}
		\end{itemize}
	\end{frame}

			\item IMGUI knihovny jsou v aktivním vývoji

			\item Implementovat v knihovnách \textit{Dear ImGui}, \textit{egui} a \textit{Qt}

	\begin{frame}[plain]\frametitle{Závěr}
		V rámci práce jsem provedl:
		\begin{itemize}
			\item Rešerši dvou přístupů ke psaní GUI aplikací: RMGUI a IMGUI
			\item Rešerši existujících IMGUI knihoven
			\item Navrhl a implementoval ukázkovou aplikaci: editor diagramů v knihovnách \textit{Dear ImGui}, \textit{egui} a \textit{Qt}			
			\item Implementoval dva benchmarky a porovnal v nich výsledné aplikace
			\item Shrnul své poznatky o zkoumaných přístupech a knihovnách, ve zkratce:		
			\begin{itemize}
				\item Přístup RMGUI je o programování callbacků, které mohou nečekaně ovlivnit chování aplikace po přidání nových funkcí.
				\item Přístup IMGUI je o programování v nekonečné smyčce, kdy je potřeba přemýšlet nad pořadím provedených operací.
				\item V RMGUI synchronizujeme stav aplikace se stavem knihovny, v IMGUI máme veškerý stav na straně aplikačního kódu.
				\item Přístup IMGUI vede na kratší kód a rychlejší vývoj.
				\item Knihovny s přístupem IMGUI oproti zavedeným RMGUI knihovnám v určitých aspektech zaostávají.
			\end{itemize}
		\end{itemize}
	\end{frame}
