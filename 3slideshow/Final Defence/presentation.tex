	% IMGUI nevýhody
	\begin{frame}[plain]\frametitle{IMGUI a textová pole}
		\begin{itemize}
			\item Textové pole je komplexní ovládací prvek, který má napříč IMGUI knihovnami omezené funkce%. Častá je např. absence horizontální posuvné lišty (\textit{scrollbar}).
			\item V knihovnách \textit{Dear ImGui} a \textit{egui}:
			\begin{itemize}
				\item Pro zvýrazňování syntaxe používám doplněk třetí strany
				\item Pro vyznačení chyby vykresluji červený obdélník nad textovým polem
			\end{itemize}
			\item Výchozí textové pole v knihovně \textit{Qt} toto podporuje
			\item Textové pole v \textit{egui} má problém s nadměrným využitím operační paměti
		\end{itemize}

			\item S přibývajícími funkcemi se zvyšuje rozštěpení logiky v aplikačním kódu
			\item Knihovna může interně spouštět callback metody v nečekaných situacích
			\begin{itemize}
				\item Např. zapnutí zvýrazňování syntaxe se bere jako změna obsahu dokumentu
				\item $\implies$ Nečekané chování, horší ladění
			\end{itemize}
			\item Některé poskytnuté metody se nechovají konzistentně
			\begin{itemize}
				\item Např. pokud uživatel táhne tlačítko, přestane se volat \texttt{mouseMoveEvent}
				\item $\implies$ Některé problémy je nutné řešit na nižší úrovni abstrakce%, na což knihovna není dělaná
			\end{itemize}

	\begin{frame}[plain, noframenumbering]
		\medskip
		„Pátral jste po důvodu, proč egui spotřebovává tolik paměti v případě použití textového pole?“
		
		\bigskip
		\begin{itemize}
			\item Knihovna připravuje na vykreslení veškerý text bez ohledu na to, jestli je aktuálně vidět
			\item Je možné přijít s vlastní logikou, která vykresluje jen to potřebné, ale pouze pro read-only textová pole
			\item Spotřeba se navíc může několikanásobně zvýšit při náhodné editaci textu uživatelem, oproti Dear ImGui je zde i vyšší spotřeba při vykreslení velkého množství ovládacích prvků (i bez textu)
			\item Autor knihovny se k tomu oficiálně nevyjádřil, ale vypadá to spíše na „důsledek robustní implementace“, nežli chybu v kódu
		\end{itemize}
