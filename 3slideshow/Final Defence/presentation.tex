	% IMGUI
	\begin{frame}[plain]\frametitle{IMGUI – Immediate mode GUI}
		\begin{itemize}
			\item „Režim okamžitého vykreslování“
			\item Žádná inicializační fáze ani callback metody, pouze hlavní smyčka			
			\item Knihovny \textit{Dear ImGui}, \textit{egui}, \textit{Nuklear}, \dots
		\end{itemize}
		\lstinputlisting[language=pseudo]{code/imgui.txt}
		\begin{itemize}
			\item Definice zmiňuje pouze tento rozdíl z pohledu programátora koncové aplikace, implementace knihoven mohou být různé.
			\item V praxi ale:
			\begin{itemize}
				\item Využití grafického API pro vykreslování ovládacích prvků (\textit{OpenGL}, \textit{DirectX}, \dots)
				\item Časté použití pro přidání ovládacích prvků do existující interaktivní aplikace
				\item Spíše GUI pro ladění vývojářem, než user-facing GUI
				\item Zmatek ohledně toho, co přesně je a není IMGUI
			\end{itemize}
		\end{itemize}
	\end{frame}

	% IMGUI nevýhody
	\begin{frame}[plain]\frametitle{IMGUI a textová pole}
		\begin{itemize}
			\item Textové pole je komplexní ovládací prvek, který má napříč IMGUI knihovnami omezené funkce%. Častá je např. absence horizontální posuvné lišty (\textit{scrollbar}).
			\item V knihovnách \textit{Dear ImGui} a \textit{egui}:
			\begin{itemize}
				\item Pro zvýrazňování syntaxe používám doplněk třetí strany
				\item Pro vyznačení chyby vykresluji červený obdélník nad textovým polem
			\end{itemize}
			\item Výchozí textové pole v knihovně \textit{Qt} toto podporuje
			\item Textové pole v \textit{egui} má problém s nadměrným využitím operační paměti
		\end{itemize}
