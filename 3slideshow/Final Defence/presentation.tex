	\begin{frame}[plain]\frametitle{IMGUI – Immediate mode GUI}
		\begin{itemize}
			\item „Režim okamžitého vykreslování“
			\item Žádná inicializační fáze ani callback metody, pouze hlavní smyčka			
			\item Knihovny \textit{Dear ImGui}, \textit{egui}, \textit{Nuklear}, \dots
		\end{itemize}
		\lstinputlisting[language=pseudo]{code/imgui.txt}
		\begin{itemize}
			\item Definice zmiňuje pouze tento rozdíl z pohledu programátora koncové aplikace, implementace knihoven mohou být různé.
			\item V praxi ale:
			\begin{itemize}
				\item Využití grafického API pro vykreslování ovládacích prvků (\textit{OpenGL}, \textit{DirectX}, \dots)
				\item Časté použití pro přidání ovládacích prvků do existující interaktivní aplikace
				\item Spíše GUI pro ladění vývojářem, než \textit{user-facing GUI}
			\end{itemize}
		\end{itemize}

	% IMGUI nevýhody
	\begin{frame}[plain]\frametitle{IMGUI a textová pole}
		\begin{itemize}
			\item Textové pole je komplexní ovládací prvek, který má napříč IMGUI knihovnami omezené funkce%. Častá je např. absence horizontální posuvné lišty (\textit{scrollbar}).
			\item V knihovnách \textit{Dear ImGui} a \textit{egui}:
			\begin{itemize}
				\item Pro zvýrazňování syntaxe používám doplněk třetí strany
				\item Pro vyznačení chyby vykresluji červený obdélník nad textovým polem
			\end{itemize}
			\item Výchozí textové pole v knihovně \textit{Qt} toto podporuje
			\item Textové pole v \textit{egui} má problém s nadměrným využitím operační paměti
		\end{itemize}

	% Otázky
	\begin{frame}[plain, noframenumbering]
		\medskip
		„Jedna z implementací ukázkového programu byla v jazyce Rust. Jsou v přístupu k tvorbě GUI v Rustu významné rozdíly oproti jiným programovacím jazykům? Uvítal byste výuku Rustu na univerzitě?“
		
		\bigskip
		\begin{itemize}
			\item Záleží na konkrétní knihovně, ale pro definici GUI je častá kompozice struktur, nebo \textit{builder pattern}:\lstinputlisting[language=rust]{code/builder.rs}	
			\item Knihovna egui kombinuje builder pattern s vnořováním uzávěrů. Obecný přístup ke tvorbě aplikace je zde podobný jako v Dear ImGui.
			\item Pro potřeby IMGUI chybí v jazyce Rust klíčové slovo \texttt{static}, takže všechna persistentní data musí být součástí nějaké struktury
			\item \textit{The roots aren't deep but the seeds are planted.}\\ — \url{https://areweguiyet.com/}
		\end{itemize}
		
	\end{frame}
	
	\begin{frame}[plain, noframenumbering]
		\medskip
		„Pátral jste po důvodu, proč egui spotřebovává tolik paměti v případě použití textového pole?“
		
		\bigskip
		\begin{itemize}
			\item \textbf{Knihovna připravuje na vykreslení veškerý text bez ohledu na to, jestli je aktuálně vidět}
			\begin{itemize}
				\item Je možné přijít s vlastní logikou, která vykresluje jen to potřebné, ale pouze pro read-only víceřádková textová pole
			\end{itemize}
			\smallskip
			\item Spotřeba se navíc může několikanásobně zvýšit při náhodné editaci textu uživatelem, oproti Dear ImGui je zde i vyšší spotřeba při vykreslení velkého množství ovládacích prvků (i těch bez textu)
			\begin{itemize}
				\item Autor knihovny se k tomu oficiálně nevyjádřil, ale vypadá to spíše na „důsledek robustní implementace“, která tento problém nebere v potaz, nežli konkrétní chybu v kódu, která by šla snadno opravit
			\end{itemize}
		\end{itemize}
		\medskip

	\end{frame}
	
	\begin{frame}[plain, noframenumbering]
		\medskip
		„Jak je to s použitím IMGUI a RMGUI přístupů na mobilních platformách?“
		
		\bigskip
		\begin{itemize}
			\item Obecně (nejen na mobilu) se nejedná o černobílou problematiku:\\ RMGUI vs. IMGUI je zjednodušený pohled
			\item Populární jsou JavaScript knihovny, kde programátor definuje \textit{komponenty}, ze kterých se skládá GUI
			\begin{itemize}				
				\item Správa stromu komponent je \textit{soběstačná}, definice komponent mohou připomínat \textit{okamžitý} režim („locality of behavior“, možnost skrýt část GUI podmíněním, \dots)
			\end{itemize}
			\item Nejčastěji používané knihovny pro tvorbu mobilních GUI také využívají přístupu s komponentami (Jetpack Compose, Flutter, React Native, \dots)
			\item Knihovny použité v této práci podporují platformy Android/iOS, ale nejsou v tomto ohledu zdaleka tak populární jako výše zmíněné
		\end{itemize}
	\end{frame}
